package frc.robot.subsystems;

import com.revrobotics.CANSparkMax;
import com.revrobotics.CANSparkBase.IdleMode;
import com.revrobotics.CANSparkLowLevel.MotorType;

public class SubstructureSubsystems {

    SubstructureMotors SubstructureMotors = new SubstructureMotors();




    public void raise() {
        SubstructureMotors.getPivot().set(0.2);
    }

    public void lower() {
        SubstructureMotors.getPivot().set(-0.2);
    }

    public void hold() {

        SubstructureMotors.getPivot().setIdleMode(IdleMode.kBrake);

    }

    public void Idle(boolean hold) {
        if(!hold) {
            SubstructureMotors.getPivot().set(0);
        }
    }

    //INCLUDE VISION METHODS FOR ANGLING SUBSTRUCTURE


}
