package frc.robot.subsystems;

import edu.wpi.first.math.geometry.Rotation2d;

import frc.robot.Constants.ModuleConstants;
import frc.robot.Constants.SubstructureConstants;;

public class SubstructureMotors {
    
    private final CANSparkMax outTop;
    private final CANSparkMax outBottom;

    private final CANSparkMax intakeMotor;

    private final CANSparkMax pivotMotor;


    public SubstructureMotors() {

        outTop = new CANSparkMax(SubstructureConstants.outTopChannel, MotorType.kBrushless);
        outBottom = new CANSparkMax(SubstructureConstants.outBottomChannel, MotorType.kBrushless);

        intakeMotor = new CANSparkMax(SubstructureConstants.intakeChannel, MotorType.kBrushless);

        pivotMotor = new CANSparkMax(SubstructureConstants.pivotChannel, MotorType.kBrushless);

    }


    public CANSparkMax getTop() {
        return outTop;
    }

    public CANSparkMax getBottom(){ 
        return outBottom;
    }

    public CANSparkMax getIntake() {
        return intakeMotor;
    }

    public CANSparkMax getPivot() {
        return pivotMotor;
    }

}
